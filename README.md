# data-kat
Необходимо создать настройку.
Заполнить данный в одинарных ковычках в файле settings.php, после загрузки проекта себе на локалбную машину или на хост.
При первой загрузке будет создана таблица с именем из $name_tbl

$user = 'root'; // ваш пользователь
$password = ''; // ваш пароль
$db = ''; // имя вашей базы данных 
$host = ''; // локальный хост
$charset = 'utf8'; // нужная кодировка
$name_tbl = ''; //имя таблицы
$login = '';
$password = '';
